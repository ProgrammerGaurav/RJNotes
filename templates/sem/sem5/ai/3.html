Tic–Tac–Toe Game using Min-Max Algorithm.
The key to the Minimax algorithm is a back and forth between the two players, where the player whose "turn it is" desires to pick the move with the maximum score. In turn, the scores for each of the available moves are determined by the opposing player deciding which of its available moves has the minimum score.
Code: 
            
                # imports
                import os
                import time

                # list store the states of the positions of mark throughout the game.
                board=[' ',' ',' ',' ',' ',' ',' ',' ',' ',' ']

                #flags for game
                players=1
                Win=1
                Draw=-1
                Running=0
                Stop=1
                Game=Running
                Mark='X'

                # This function prints out the board that it was passed."board" is a list of 10 strings representing the board (ignore index 0)
                def DrawBoard():
                    print(" %c | %c | %c " % (board[1],board[2],board[3]))
                    print("___|___|___")
                    print(" %c | %c | %c " % (board[4],board[5],board[6]))
                    print("___|___|___")
                    print(" %c | %c | %c " % (board[7],board[8],board[9]))
                    print("   |   |   ")

                # This functions checks whether the position is already marked or not.
                def CheckPosition(x):
                    if(board[x] == ' '):
                        return True
                    else:
                        return False

                # This function checks whether the game is won by any one of the player or not.
                def CheckWin():
                    global Game

                    # Checks for winning condition horizontally
                    if (board[1] == board[2] and board[2] == board[3] and board[1] != ' '):
                        Game = Win
                    elif (board[4] == board[5] and board[5] == board[6] and board[4] != ' '):
                        Game = Win
                    elif (board[7] == board[8] and board[8] == board[9] and board[7] != ' '):
                        Game = Win

                    # Checks for winning condition vertically
                    elif (board[1] == board[4] and board[4] == board[7] and board[1] != ' '):
                        Game = Win
                    elif (board[2] == board[5] and board[5] == board[8] and board[2] != ' '):
                        Game = Win
                    elif (board[3] == board[6] and board[6] == board[9] and board[3] != ' '):
                        Game = Win

                    # Checks for winning condition diagonally
                    elif (board[1] == board[5] and board[5] == board[9] and board[1] != ' '):
                        Game = Win
                    elif (board[3] == board[5] and board[5] == board[7] and board[3] != ' '):
                        Game = Win

                    # Checks if all the postions are marked or not if true then game is a draw
                    elif (board[1] != ' 'and board[2] != ' ' and board[3] != ' ' and board[4] != ' ' and board[5] != ' ' and board[6] != ' ' and board[7] != ' ' and board[8] != ' ' and board[9] != ' '):
                        Game = Draw

                    # if none of the above conditions are true then game is still running
                    else:
                        Game=Running

                # Driver code
                print("Tic-tac-toe Game")
                print("Player 1[X]---Player 2[O]/n")
                print()
                print()
                print("Please Wait.....")
                time.sleep(1)

                # While Game is running
                while(Game == Running):
                    #os.system('cls')

                    # Draw the current state of the board
                    DrawBoard()

                    # Check which player’s turn it is
                    if(players%2 !=0):
                        print("Player 1's chance:")
                        Mark='x'
                    else:
                        print("Player 2's chance:")
                        Mark='o'
                    # Ask player for position on the board to make a mark
                    choice=int(input("Enter the position between[1-9]:"))
                    # Check if  the entered position is already marked or not
                    if(CheckPosition(choice)):
                        # if position is empty the mark the position
                        board[choice]=Mark
                        # increment players to mark that the current player has played it's turn
                        players+=1
                        # Check if player has won or not
                        CheckWin()
                # clears command prompt screen
                #os.system('cls')

                # Draw current board
                DrawBoard()

                #print the result of the game
                if(Game == Draw):
                    print("Game Draw")
                elif(Game == Win):
                    players-=1
                    if(players%2 != 0):
                        print("Player 1 won")
                    else:
                        print("Player 2 won")
