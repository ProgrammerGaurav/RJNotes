                def tweetdf():
                    # Initializing the DataFrame
                    df = pd.DataFrame(data=[tweet['text']
                                            for tweet in tweets.tweets], columns=['tweets'])
                    # adding columns to the DataFrame using numpy.array()
                    df['id'] = np.array([tweet['id'] for tweet in tweets.tweets])
                    df['len'] = np.array([len(tweet['text']) for tweet in tweets.tweets])
                    df['date'] = np.array([tweet['created_at'] for tweet in tweets.tweets])
                    df['likes'] = np.array([tweet['favourite_count']
                                            for tweet in tweets.tweets])
                    df['retweets'] = np.array([tweet['retweet_count']
                                            for tweet in tweets.tweets])
                    return df
