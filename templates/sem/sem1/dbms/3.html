Simple MySQL Queries
Creating a table employee
Code: 
            
                create table employee(
                    id int NOT NULL,
                    Name varchar(20) NOT NULL,
                    salary int NOT NULL,
                    PRIMARY KEY(id)
                );
Output: 
            
                Query OK, 0 rows affected (0.07 sec)
Inserting values into employee
Code: 
            
                insert into employee values(1, "Gaurav", 1200000);
                insert into employee values(2, "Harshad", 1200000);
                insert into employee values(3, "ABC", 120000);
                insert into employee values(4, "PQR", 100000);
