1) Date Functions
adddate()
Code: 
            
                select adddate("1988-01-02",31);
Output: 
            
                +--------------------------+
                | adddate("1988-01-02",31) |
                +--------------------------+
                | 1988-02-02               |
                +--------------------------+
datediff()
Code: 
            
                select datediff("1988-01-05","1988-01-03");
Output: 
            
                +-------------------------------------+
                | datediff("1988-01-05","1988-01-03") |
                +-------------------------------------+
                |                                   2 |
                +-------------------------------------+
month()
Code: 
            
                select month("1988-01-02");
Output: 
            
                +---------------------+
                | month("1988-01-02") |
                +---------------------+
                |                   1 |
                +---------------------+
year()
Code: 
            
                select year("1988-01-02");
Output: 
            
                +--------------------+
                | year("1988-01-02") |
                +--------------------+
                |               1988 |
                +--------------------+
hour()
Code: 
            
                select hour("10:05:03");
Output: 
            
                +------------------+
                | hour("10:05:03") |
                +------------------+
                |               10 |
                +------------------+
minute()
Code: 
            
                select minute("10:05:03");
Output: 
            
                +--------------------+
                | minute("10:05:03") |
                +--------------------+
                |                  5 |
                +--------------------+

Code:

                +--------------------+
                | second("10:05:03") |
                +--------------------+
                |                  3 |
                +--------------------+

Code: 
            
                select now();
Output: 
            
                +---------------------+
                | now()               |
                +---------------------+
